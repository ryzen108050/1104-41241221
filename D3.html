D3.js 套件

套件名稱

D3.js

官方網站連結

https://d3js.org/

範例程式碼

<script src="https://d3js.org/d3.v7.min.js"></script>
<script>
  const data = [30, 86, 168, 234, 24, 34];
  const width = 400;
  const height = 100;
  const svg = d3.select('svg')
    .attr('width', width)
    .attr('height', height);
  svg.selectAll('rect')
    .data(data)
    .enter().append('rect')
    .attr('width', (d) => d)
    .attr('height', 20)
    .attr('y', (d, i) => i * 25)
    .attr('fill', 'steelblue');
  </script>
<svg></svg>
