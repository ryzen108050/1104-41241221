Lottie 動畫套件

套件名稱

Lottie

官方網站連結

https://airbnb.io/lottie/

範例程式碼

<!-- 引入 Lottie -->
<script src="https://cdnjs.cloudflare.com/ajax/libs/bodymovin/5.7.5/lottie.min.js"></script>

<!-- Lottie 動畫容器 -->
<div id="lottie-animation" style="width: 100%; height: 200px;"></div>

<script>
  const animation = lottie.loadAnimation({
    container: document.getElementById('lottie-animation'),
    renderer: 'svg',
    loop: true,
    autoplay: true,
    path: 'https://assets7.lottiefiles.com/packages/lf20_jcikwtux.json' // Lottie 動畫檔案
  });
</script>
